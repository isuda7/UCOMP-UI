<!-- Open Trigger --> <button type="button" class="btn module-*" data-bs-toggle="popover" data-bs-custom-class="module-a" data-bs-container="body" data-bs-trigger="hover focus" data-bs-title="Custom popover" data-bs-content="This popover is themed via CSS variables."> <span>Custom popover</span> </button> <!-- Init Script --> <script> // Case1. 기본호출 document.addEventListener("DOMContentLoaded", function() { const popover = new bootstrap.Popover('#examplePopover1'); }); // Case2. 공통호출 function setPopover(selector){ const popoverTriggerList = document.querySelectorAll('[data-bs-toggle="popover"]'); const popoverList = [...popoverTriggerList].map(popoverTriggerEl => { return new bootstrap.Popover(popoverTriggerEl); }); } </script>

사용자정의

<!-- 사용자정의 속성 --> <button class="btn" data-template-id="tpl-demo-1"> 템플릿 팝오버 </button> <!-- 사용자정의 HTML --> <template id="tpl-demo-1"> <div> <strong>강조 텍스트</strong> <hr /> <a href="#">자세히 보기</a> </div> </template> // Case3. 공통호출(사용자정의 추가) function setPopover() { const popoverTriggerList = document.querySelectorAll('[data-bs-toggle="popover"]'); const popoverList = [...popoverTriggerList].map((el) => { // 사용자정의 HTML 구조 정의 const templateId = el.dataset.templateId; if (templateId) { const template = document.getElementById(templateId); if (template) { return new bootstrap.Popover(el, { content: template.innerHTML, html: true, }); } } // 기본 popover (data-bs-content / data-bs-title 등으로 처리되는 경우) return new bootstrap.Popover(el); }); }
